177 {
178  static std::chrono::_V2::system_clock::time_point t_start;
179 
180  if (mode==0)
181  {
182  t_start = std::chrono::high_resolution_clock::now();
183  if (s != "")
184  std::cout << "PROCESS: " << s << std::endl;
185  return 0.0;
186  }
187  else {
188  auto t_end = std::chrono::high_resolution_clock::now();
189  //std::cout << "Elapsed time for process " << s << " is: " << (t_end-t_start).count()*1E-9 << " seconds\n";
190  if (scale == T_S)
191  {
192  std::cout << "END: Elapsed time for process " << s << " is: " << std::chrono::duration_cast<std::chrono::seconds>(t_end-t_start).count() << " seconds\n\n";
193  return std::chrono::duration_cast<std::chrono::seconds>(t_end-t_start).count();
194  }
195  else if (scale == T_MS)
196  {
197  std::cout << "END: Elapsed time for process " << s << " is: " << std::chrono::duration_cast<std::chrono::milliseconds>(t_end-t_start).count() << " milliseconds\n\n";
198  return std::chrono::duration_cast<std::chrono::milliseconds>(t_end-t_start).count();
199  }
200  else
201  {
202  std::cout << "END: Elapsed time for process " << s << " is: " << std::chrono::duration_cast<std::chrono::microseconds>(t_end-t_start).count() << " microseconds\n\n";
203  return std::chrono::duration_cast<std::chrono::microseconds>(t_end-t_start).count();
